#include <iostream>
#include <vector>
#include <string>
#include <iomanip>
#include <limits>
#include <algorithm>
#include <ctime>

// --- ANSI Color Codes ---
const std::string COLOR_GREEN = "\033[32m";
const std::string COLOR_RED = "\033[31m";
const std::string COLOR_YELLOW = "\033[33m";
const std::string COLOR_CYAN = "\033[36m";
const std::string COLOR_RESET = "\033[0m";

// --- 1. Product Class (Unchanged) ---
class Product {
public:
    int id;
    std::string name;
    double price;
    int stock;

    Product(int _id, const std::string& _name, double _price, int _stock)
        : id(_id), name(_name), price(_price), stock(_stock) {}

    void displayInventoryRow() const {
        std::cout << std::fixed << std::setprecision(2);
        std::cout << "| " << std::setw(5) << id
                  << " | " << std::setw(28) << std::left << name
                  << " | $" << std::setw(9) << std::right << price
                  << " | " << std::setw(8) << stock << " |\n";
    }
};

// --- 2. Customer Class (UPDATED with isRegular) ---
class Customer {
public:
    int id;
    std::string name;
    std::string contact;
    bool isRegular; // New field for loyalty status

    Customer(int _id, const std::string& _name, const std::string& _contact, bool _isRegular = false)
        : id(_id), name(_name), contact(_contact), isRegular(_isRegular) {}

    void displayCustomerRow() const {
        std::string status = isRegular ? "YES" : "NO";
        std::cout << "| " << std::setw(5) << id
                  << " | " << std::setw(25) << std::left << name
                  << " | " << std::setw(20) << contact 
                  << " | " << std::setw(8) << status << " |\n";
    }
};

// --- 3. Sale Item Structure (Unchanged) ---
struct SaleItem {
    int productId;
    std::string productName;
    double unitPrice;
    int quantity;
    double lineTotal;

    void displayBillRow() const {
        std::cout << std::fixed << std::setprecision(2);
        std::cout << std::setw(3) << quantity 
                  << " x " << std::setw(25) << std::left << productName 
                  << " @ $" << std::setw(8) << std::right << unitPrice 
                  << " = $" << std::setw(10) << lineTotal << "\n";
    }
};

// --- 4. Shop Management Class ---
class Shop {
private:
    std::vector<Product> inventory;
    std::vector<Customer> customers;
    std::vector<SaleItem> currentSaleCart;
    double salesTaxRate = 0.08;
    int nextProductId = 1005;
    int nextCustomerId = 3; 

public:
    Shop() {
        // Initialize sample data
        inventory.emplace_back(1001, "Milk (1L)", 3.99, 50);
        inventory.emplace_back(1002, "Bread (Whole Wheat)", 4.50, 30);
        inventory.emplace_back(1003, "Eggs (Dozen)", 7.25, 12);
        inventory.emplace_back(1004, "Ground Coffee (500g)", 15.00, 25);
        customers.emplace_back(1, "John Smith", "555-0001", false);
        customers.emplace_back(2, "Alice Jones", "555-0002", true);
    }
    
    // --- Utility Functions ---

    void clearInputBuffer() {
        if (std::cin.fail()) {
            std::cin.clear();
            std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        }
        std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
    }
    
    // --- Display Functions ---

    void displayInventory() const {
        std::cout << "\n" << COLOR_YELLOW << "--- CURRENT PRODUCT INVENTORY ---\n" << COLOR_RESET;
        std::cout << "===============================================================\n";
        std::cout << "| ID    | Product Name                | Price      | Stock    |\n";
        std::cout << "===============================================================\n";
        for (const auto& product : inventory) {
            product.displayInventoryRow();
        }
        std::cout << "===============================================================\n";
    }
    
    void displayCustomers() const {
        std::cout << "\n" << COLOR_YELLOW << "--- CUSTOMER LIST ---\n" << COLOR_RESET;
        std::cout << "============================================================\n";
        std::cout << "| ID    | Name                      | Contact            | Regular |\n";
        std::cout << "============================================================\n";
        for (const auto& customer : customers) {
            customer.displayCustomerRow();
        }
        std::cout << "============================================================\n";
    }

    // --- Customer Self-Registration Function ---
    void registerNewCustomer() {
        std::string name, contact;
        std::cout << COLOR_GREEN << "\n--- CUSTOMER REGISTRATION ---\n" << COLOR_RESET;
        std::cout << "Enter your Full Name: ";
        std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        std::getline(std::cin, name);
        
        std::cout << "Enter Contact (Phone/Email): ";
        std::getline(std::cin, contact);

        customers.emplace_back(nextCustomerId, name, contact, false);
        std::cout << COLOR_GREEN << "Registration successful! Your Customer ID is: " << nextCustomerId << COLOR_RESET << "\n";
        std::cout << "Please use this ID to identify yourself during checkout.\n";
        nextCustomerId++;
    }

    // --- Admin Functions ---

    void addProduct() {
        std::string name;
        double price;
        int stock;
        
        std::cout << COLOR_CYAN << "\n--- ADD NEW PRODUCT ---\n" << COLOR_RESET;
        std::cout << "Enter Product Name: ";
        std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        std::getline(std::cin, name);
        
        std::cout << "Enter Price: $";
        if (!(std::cin >> price)) return;
        
        std::cout << "Enter Initial Stock: ";
        if (!(std::cin >> stock)) return;

        inventory.emplace_back(nextProductId, name, price, stock);
        std::cout << COLOR_GREEN << "Product added successfully! ID: " << nextProductId << COLOR_RESET << "\n";
        nextProductId++;
    }
    
    void deleteProduct() {
        int id;
        std::cout << COLOR_CYAN << "\n--- DELETE PRODUCT ---\n" << COLOR_RESET;
        std::cout << "Enter Product ID to delete: ";
        if (!(std::cin >> id)) return;

        auto it = std::remove_if(inventory.begin(), inventory.end(), 
                               [&](const Product& p){ return p.id == id; });

        if (it != inventory.end()) {
            inventory.erase(it, inventory.end());
            std::cout << COLOR_GREEN << "Product ID " << id << " deleted successfully.\n" << COLOR_RESET;
        } else {
            std::cout << COLOR_RED << "Error: Product ID not found.\n" << COLOR_RESET;
        }
    }
    
    // --- Customer Shopping Functions (UPDATED MENU) ---

    void customerShoppingMode() {
        currentSaleCart.clear();
        int choice;
        
        do {
            std::cout << COLOR_GREEN << "\n--- CUSTOMER SHOPPING MENU ---\n" << COLOR_RESET;
            std::cout << "1. View Products\n";
            std::cout << "2. Add Product to Cart\n";
            std::cout << "3. View/Modify Cart\n";
            std::cout << "4. Checkout\n";
            std::cout << "5. Register New Customer\n"; // NEW OPTION
            std::cout << "6. Return to Main Login\n"; // SHIFTED OPTION
            std::cout << "Enter choice: ";

            if (!(std::cin >> choice)) {
                std::cout << COLOR_RED << "Invalid input.\n" << COLOR_RESET;
                clearInputBuffer();
                continue;
            }

            switch(choice) {
                case 1:
                    displayInventory();
                    break;
                case 2:
                    addToCartLogic();
                    break;
                case 3:
                    viewAndModifyCart();
                    break;
                case 4:
                    if (!currentSaleCart.empty()) {
                        generateBill();
                        return; // Exit shopping mode after successful checkout
                    } else {
                        std::cout << COLOR_YELLOW << "Cart is empty. Please add items first.\n" << COLOR_RESET;
                    }
                    break;
                case 5:
                    registerNewCustomer(); // Added register function call
                    break;
                case 6:
                    std::cout << COLOR_YELLOW << "Returning to main login menu.\n" << COLOR_RESET;
                    break;
                default:
                    std::cout << COLOR_RED << "Invalid option.\n" << COLOR_RESET;
            }
            if (choice != 6 && choice != 4) {
                std::cout << "\nPress ENTER to continue...";
                clearInputBuffer();
            }

        } while (choice != 6);
    }
    
    void addToCartLogic() {
        int id, quantity;
        
        displayInventory();
        std::cout << "\nEnter Product ID to add: ";
        if (!(std::cin >> id)) return;
        
        // Find the product
        auto it = std::find_if(inventory.begin(), inventory.end(), 
                               [&](const Product& p){ return p.id == id; });

        if (it == inventory.end()) {
            std::cout << COLOR_RED << "Error: Product ID not found.\n" << COLOR_RESET;
            return;
        }

        Product& product = *it;

        std::cout << "Enter Quantity for " << product.name << ": ";
        if (!(std::cin >> quantity) || quantity <= 0) {
            std::cout << COLOR_RED << "Invalid quantity.\n" << COLOR_RESET;
            return;
        }

        // Check stock
        int currentCartQuantity = 0;
        auto cart_it_check = std::find_if(currentSaleCart.begin(), currentSaleCart.end(), 
                                        [&](const SaleItem& item){ return item.productId == id; });
        if (cart_it_check != currentSaleCart.end()) {
            currentCartQuantity = cart_it_check->quantity;
        }

        if (product.stock < (currentCartQuantity + quantity)) {
            std::cout << COLOR_RED << "Error: Stock limit reached. Only " << product.stock << " units in stock total.\n" << COLOR_RESET;
            return;
        }
        
        // --- Process Sale Item (Immediate Stock Update) ---
        
        // 1. Update Cart
        double lineTotal = product.price * quantity;

        if (cart_it_check != currentSaleCart.end()) {
            cart_it_check->quantity += quantity;
            cart_it_check->lineTotal += lineTotal;
        } else {
            currentSaleCart.push_back({
                product.id,
                product.name,
                product.price,
                quantity,
                lineTotal
            });
        }
        
        // 2. Reduce Stock (since it passed the check)
        product.stock -= quantity;
        
        std::cout << COLOR_GREEN << "Added " << quantity << " x " << product.name << " to cart successfully." << COLOR_RESET << "\n";
    }

    void viewAndModifyCart() {
        if (currentSaleCart.empty()) {
            std::cout << "\nYour cart is empty.\n";
            return;
        }

        std::cout << COLOR_YELLOW << "\n--- YOUR SHOPPING CART ---\n" << COLOR_RESET;
        std::cout << "--------------------------------------------------------\n";
        std::cout << "ID | Product Name            | Qty | Price     | Total   \n";
        std::cout << "--------------------------------------------------------\n";
        
        double subtotal = 0.0;
        for (const auto& item : currentSaleCart) {
            std::cout << std::setw(2) << item.productId 
                      << " | " << std::setw(25) << std::left << item.productName 
                      << " | " << std::setw(3) << item.quantity 
                      << " | $" << std::setw(8) << std::right << item.unitPrice 
                      << " | $" << std::setw(8) << item.lineTotal << "\n";
            subtotal += item.lineTotal;
        }
        std::cout << "--------------------------------------------------------\n";
        std::cout << std::setw(50) << std::right << "Subtotal: $" << std::setw(8) << subtotal << "\n";
        
        int choice;
        std::cout << "\nOptions:\n";
        std::cout << "1. Delete an item\n";
        std::cout << "2. Change quantity of an item\n";
        std::cout << "3. Continue Shopping\n";
        std::cout << "Enter choice: ";
        
        if (!(std::cin >> choice)) return;

        int id;
        
        switch(choice) {
            case 1:
                std::cout << "Enter Product ID to delete: ";
                if (!(std::cin >> id)) return;
                deleteFromCart(id);
                break;
            case 2:
                std::cout << "Enter Product ID to change quantity: ";
                if (!(std::cin >> id)) return;
                changeQuantityInCart(id);
                break;
            case 3:
                break; // Continue shopping
            default:
                std::cout << COLOR_RED << "Invalid option.\n" << COLOR_RESET;
        }
    }
    
    void deleteFromCart(int id) {
        auto cart_it = std::find_if(currentSaleCart.begin(), currentSaleCart.end(), 
                                    [&](const SaleItem& item){ return item.productId == id; });

        if (cart_it != currentSaleCart.end()) {
            // Restore stock (Important!)
            auto product_it = std::find_if(inventory.begin(), inventory.end(), 
                                           [&](const Product& p){ return p.id == id; });
            if (product_it != inventory.end()) {
                product_it->stock += cart_it->quantity;
            }

            currentSaleCart.erase(cart_it);
            std::cout << COLOR_GREEN << "Item deleted from cart and stock restored.\n" << COLOR_RESET;
        } else {
            std::cout << COLOR_RED << "Product ID not found in cart.\n" << COLOR_RESET;
        }
    }

    void changeQuantityInCart(int id) {
        int newQuantity;
        auto cart_it = std::find_if(currentSaleCart.begin(), currentSaleCart.end(), 
                                    [&](const SaleItem& item){ return item.productId == id; });

        if (cart_it == currentSaleCart.end()) {
            std::cout << COLOR_RED << "Product ID not found in cart.\n" << COLOR_RESET;
            return;
        }
        
        auto product_it = std::find_if(inventory.begin(), inventory.end(), 
                                       [&](const Product& p){ return p.id == id; });
        if (product_it == inventory.end()) return;

        std::cout << "Current Quantity: " << cart_it->quantity << ". Enter new quantity (0 to delete): ";
        if (!(std::cin >> newQuantity) || newQuantity < 0) {
            std::cout << COLOR_RED << "Invalid quantity.\n" << COLOR_RESET;
            return;
        }

        if (newQuantity == 0) {
            deleteFromCart(id);
            return;
        }

        int stockChange = newQuantity - cart_it->quantity;

        // Check if increasing quantity is possible with current stock
        if (stockChange > 0 && stockChange > product_it->stock) {
            std::cout << COLOR_RED << "Error: Only " << product_it->stock << " units available to increase.\n" << COLOR_RESET;
            return;
        }

        // Apply changes
        product_it->stock -= stockChange;
        cart_it->quantity = newQuantity;
        cart_it->lineTotal = cart_it->unitPrice * newQuantity;

        std::cout << COLOR_GREEN << "Quantity updated successfully.\n" << COLOR_RESET;
    }

    // --- Checkout/Billing Function ---

    void generateBill() {
        if (currentSaleCart.empty()) {
            std::cout << COLOR_RED << "Cannot generate bill. Cart is empty.\n" << COLOR_RESET;
            return;
        }
        
        // **Identify Customer**
        int customerId = 0;
        std::cout << "\nEnter Customer ID (or 0 for guest checkout): ";
        if (!(std::cin >> customerId)) return;
        
        double subtotal = 0.0;
        
        std::time_t t = std::time(0);
        std::tm* now = std::localtime(&t);

        std::cout << "\n\n" << COLOR_YELLOW << "=============================================\n" << COLOR_RESET;
        std::cout << "            " << COLOR_GREEN << "OFFICIAL SALES RECEIPT" << COLOR_RESET << "           \n";
        std::cout << "=============================================\n";
        
        // Display customer details if registered
        auto customer_it = std::find_if(customers.begin(), customers.end(), 
                                        [&](const Customer& c){ return c.id == customerId; });
        
        if (customer_it != customers.end()) {
            std::cout << "Customer: " << customer_it->name 
                      << " (ID: " << customer_it->id 
                      << (customer_it->isRegular ? ", Regular)" : ")") << "\n";
        } else {
             std::cout << "Customer: Guest\n";
        }

        std::cout << "Date: " << (now->tm_year + 1900) << "-" 
                  << (now->tm_mon + 1) << "-" << now->tm_mday 
                  << " | Time: " << now->tm_hour << ":" << now->tm_min << "\n";
        std::cout << "---------------------------------------------\n";
        
        for (const auto& item : currentSaleCart) {
            item.displayBillRow();
            subtotal += item.lineTotal;
        }
        std::cout << "---------------------------------------------\n";

        double taxAmount = subtotal * salesTaxRate;
        double grandTotal = subtotal + taxAmount;

        std::cout << std::fixed << std::setprecision(2);
        std::cout << std::setw(38) << std::right << "Subtotal: $" << std::setw(10) << subtotal << "\n";
        std::cout << std::setw(38) << std::right << "Tax (" << (salesTaxRate * 100) << "%): $" << std::setw(10) << taxAmount << "\n";
        std::cout << COLOR_YELLOW << std::setw(38) << std::right << "GRAND TOTAL: $" << std::setw(10) << grandTotal << COLOR_RESET << "\n";
        std::cout << "=============================================\n";
        std::cout << "        Thank you for shopping with us!      \n";
        std::cout << "=============================================\n\n";
        
        // Clear cart for the next customer
        currentSaleCart.clear();
    }
};

// --- Main Menus ---

void adminMenu(Shop& manager) {
    int choice;
    do {
        std::cout << COLOR_CYAN << "\n\t--- ADMIN MANAGEMENT MENU ---\n" << COLOR_RESET;
        std::cout << "1. View Products\n";
        std::cout << "2. Add New Product\n";
        std::cout << "3. Delete Product\n";
        std::cout << "4. View Customer List (Incl. Regular Status)\n";
        std::cout << "5. Return to Main Login\n";
        std::cout << "Enter your choice: ";

        if (!(std::cin >> choice)) {
            std::cout << COLOR_RED << "Invalid input.\n" << COLOR_RESET;
            manager.clearInputBuffer();
            continue;
        }

        switch (choice) {
            case 1: manager.displayInventory(); break;
            case 2: manager.addProduct(); break;
            case 3: 
                manager.displayInventory();
                manager.deleteProduct(); 
                break;
            case 4: manager.displayCustomers(); break;
            case 5: std::cout << COLOR_YELLOW << "Logging out of Admin mode.\n" << COLOR_RESET; break;
            default: std::cout << COLOR_RED << "Invalid choice. Please try again.\n" << COLOR_RESET; break;
        }

        if (choice != 5) {
            std::cout << "\nPress ENTER to continue...";
            manager.clearInputBuffer();
        }
    } while (choice != 5);
}

// --- Main Execution (UPDATED MENU) ---
int main() {
    Shop shopManager;
    int choice;
    
    // Simple screen clear for a cleaner menu (optional)
    auto clearScreen = []() {
        #ifdef _WIN32
            system("cls");
        #else
            system("clear");
        #endif
    };

    do {
        clearScreen();
        std::cout << COLOR_GREEN << "\n\t--- SHOP MANAGEMENT SYSTEM LOGIN ---\n" << COLOR_RESET;
        std::cout << "1. Login as Admin\n";
        std::cout << "2. Shop as Customer\n";
        std::cout << "3. Exit System\n"; // Shifted back to 3
        std::cout << "Enter your choice: ";

        if (!(std::cin >> choice)) {
            std::cout << COLOR_RED << "Invalid input. Please enter a number.\n" << COLOR_RESET;
            shopManager.clearInputBuffer();
            continue;
        }

        switch (choice) {
            case 1:
                adminMenu(shopManager);
                break;
            case 2:
                shopManager.customerShoppingMode();
                break;
            case 3:
                std::cout << "System shutting down. Goodbye!\n";
                break;
            default:
                std::cout << COLOR_RED << "Invalid choice. Please try again.\n" << COLOR_RESET;
                shopManager.clearInputBuffer();
                break;
        }
        
    } while (choice != 3); // Back to 3

    return 0;
}
